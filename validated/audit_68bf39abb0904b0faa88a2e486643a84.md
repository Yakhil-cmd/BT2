### Summary

The `ExitToNear` precompile optimistically proceeds with bridging ETH (base token) or ERC-20 tokens to a user-supplied NEAR account ID without verifying that the destination account is registered for NEP-141 fungible-token storage. When the `error_refund` compile-time feature is disabled, a failed `ft_transfer` NEAR promise has no attached callback, so the EVM-side burn is irreversible and the user's funds are permanently lost.

fn parse_recipient(recipient: &[u8]) -> Result<Recipient<'_>, ExitError> {
    let recipient = str::from_utf8(recipient)
        .map_err(|_| ExitError::Other(Cow::from("ERR_INVALID_RECEIVER_ACCOUNT_ID")))?;
    let (receiver_account_id, message) = recipient.split_once(':').map_or_else(
        || (recipient, None),
        |(recipient, msg)| {
            if msg == UNWRAP_WNEAR_MSG {
                (recipient, Some(Message::UnwrapWnear))
            } else {
                (recipient, Some(Message::Omni(msg)))
            }
        },
    );

    Ok(Recipient {
        receiver_account_id: receiver_account_id
            .parse()
            .map_err(|_| ExitError::Other(Cow::from("ERR_INVALID_RECEIVER_ACCOUNT_ID")))?,
        message,
    })
}
