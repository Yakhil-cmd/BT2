### Title
Stale Delegated Permissions Persist After `CronosAdmin` Rotation, Enabling Unauthorized Token Mapping Corruption — (`x/cronos/keeper/permissions.go`, `x/cronos/keeper/msg_server.go`)

---

### Summary

The `x/cronos` module stores the privileged admin address (`CronosAdmin`) in the `Params` KV store and stores delegated per-address permissions (`CanChangeTokenMapping`, `CanTurnBridge`) in a **separate** KV store under `prefixAdminToPermissions`. When governance rotates `CronosAdmin` via `MsgUpdateParams`, the `prefixAdminToPermissions` store is never cleared. Any address previously granted `CanChangeTokenMapping` by the old admin retains that permission indefinitely and can continue to corrupt token mappings — bypassing the governance-intended revocation.

---

### Finding Description

**Two independent state stores govern the same authorization surface:**

`CronosAdmin` lives in the `Params` blob: [1](#0-0) 

Delegated permissions live in a separate prefix: [2](#0-1) 

`HasPermission` reads both stores independently — first checking whether the caller equals the current `CronosAdmin` from params, then checking the `prefixAdminToPermissions` store: [3](#0-2) 

`MsgUpdatePermissions` (admin-only) writes to `prefixAdminToPermissions`: [4](#0-3) 

`MsgUpdateParams` (governance-only) writes only to the `Params` blob and **never touches `prefixAdminToPermissions`**: [5](#0-4) 

There is no `GetAllPermissions` keeper method or gRPC query to enumerate all permissioned addresses. The only query is per-address: [6](#0-5) 

**The implicit invariant** — that all `CanChangeTokenMapping` holders are trusted by the current admin — is never enforced at the protocol level. The two stores can diverge silently.

---

### Impact Explanation

An address holding a stale `CanChangeTokenMapping` bit can call `MsgUpdateTokenMapping` and pass the permission check: [7](#0-6) 

`RegisterOrUpdateTokenMapping` will then remap any IBC or gravity denom to an attacker-controlled contract, or delete an existing mapping: [8](#0-7) 

Once a denom is remapped, `ConvertVouchersToEvmCoins` routes all incoming IBC vouchers for that denom through the attacker's contract: [9](#0-8) 

This is a **High** impact: direct corruption of token mappings and denom/contract bindings with security impact, and a bypass of governance authority over the permission system.

---

### Likelihood Explanation

The scenario requires the old `CronosAdmin` to have previously delegated `CanChangeTokenMapping` to at least one address. This is the intended operational use of the permission system (ADR-009 explicitly designs for it). When governance rotates `CronosAdmin` — the expected security response to a compromised admin — there is no protocol mechanism to revoke delegated permissions, and no enumeration query to discover which addresses hold them. The new admin must know to manually revoke each address, but has no on-chain tool to find them.

---

### Recommendation

`MsgUpdateParams` (or `SetParams`) should atomically clear the entire `prefixAdminToPermissions` store whenever `CronosAdmin` changes. Alternatively, add an explicit `GetAllPermissions` iterator and require governance to pass a revocation list alongside any admin rotation. The invariant — "all delegated permissions are authorized by the current admin" — must be enforced at the state-transition boundary, not left as an implicit operational requirement.

---

### Proof of Concept

1. Governance sets `CronosAdmin = addrA` via `MsgUpdateParams`.
2. `addrA` calls `MsgUpdatePermissions(address=addrX, permissions=1)` → `addrX` gets `CanChangeTokenMapping` written to `prefixAdminToPermissions`.
3. Governance detects `addrA` is compromised; passes `MsgUpdateParams` setting `CronosAdmin = addrB`. `prefixAdminToPermissions` is untouched.
4. `addrX` submits `MsgUpdateTokenMapping(denom="ibc/ATOM...", contract=<malicious>)`.
5. `HasPermission` reads `CronosAdmin = addrB` (no match), then reads `GetPermissions(addrX) = 1`, passes the `CanChangeTokenMapping` check.
6. `RegisterOrUpdateTokenMapping` writes the new denom→malicious-contract mapping to the KV store.
7. All subsequent `OnRecvVouchers` calls for `ibc/ATOM` route through the malicious contract, corrupting user balances. [3](#0-2) [5](#0-4)

```go
func (k Keeper) HasPermission(ctx sdk.Context, accounts []sdk.AccAddress, permissionsToCheck uint64) bool {
	// case when no permission is needed
	if permissionsToCheck == 0 {
		return true
	}
	admin := k.GetParams(ctx).CronosAdmin
	for _, account := range accounts {
		if admin == account.String() {
			return true
		}
		permission := k.GetPermissions(ctx, account)
		if permission&permissionsToCheck == permissionsToCheck {
			return true
		}
	}

	return false
```

```go
		return nil, err
	}
```

```go
func (k msgServer) UpdatePermissions(goCtx context.Context, msg *types.MsgUpdatePermissions) (*types.MsgUpdatePermissionsResponse, error) {
	ctx := sdk.UnwrapSDKContext(goCtx)
	admin := k.Keeper.GetParams(ctx).CronosAdmin
	// if admin is empty, no sender could be equal to it
	if admin != msg.From {
		return nil, errors.Wrap(sdkerrors.ErrUnauthorized, "msg sender is not authorized")
	}
	acc, err := sdk.AccAddressFromBech32(msg.Address)
	if err != nil {
		return nil, err
	}
	k.SetPermissions(ctx, acc, msg.Permissions)

	return &types.MsgUpdatePermissionsResponse{}, nil
```

```go
// Permissions returns the permissions of a specific account
func (k Keeper) Permissions(goCtx context.Context, req *types.QueryPermissionsRequest) (*types.QueryPermissionsResponse, error) {
	if req == nil {
		return nil, status.Error(codes.InvalidArgument, "empty request")
	}
	ctx := sdk.UnwrapSDKContext(goCtx)
	acc, err := sdk.AccAddressFromBech32(req.Address)
	if err != nil {
		return nil, err
	}
	admin := k.GetParams(ctx).CronosAdmin
	if admin == acc.String() {
		return &types.QueryPermissionsResponse{
			CanChangeTokenMapping: true,
			CanTurnBridge:         true,
		}, nil
	}
	permissions := k.GetPermissions(ctx, acc)
	return &types.QueryPermissionsResponse{
		CanChangeTokenMapping: CanChangeTokenMapping == (permissions & CanChangeTokenMapping),
		CanTurnBridge:         CanTurnBridge == (permissions & CanTurnBridge),
	}, nil
}
```

```go
func (k Keeper) RegisterOrUpdateTokenMapping(ctx sdk.Context, msg *types.MsgUpdateTokenMapping) error {
	if types.IsSourceCoin(msg.Denom) {
		_, err := types.GetContractAddressFromDenom(msg.Denom)
		if err != nil {
			return err
		}

		if !common.IsHexAddress(msg.Contract) {
			return errors.Wrapf(sdkerrors.ErrInvalidAddress, "invalid contract address (%s)", msg.Contract)
		}
		contract := common.HexToAddress(msg.Contract)
		if err := k.ensureContractCode(ctx, contract); err != nil {
			return err
		}
		if err := validateContractAddressForSourceDenom(msg.Denom, contract, true); err != nil {
			return err
		}

		if err := k.SetExternalContractForDenom(ctx, msg.Denom, contract); err != nil {
			return err
		}

		// check that the coin is registered, otherwise register it
		metadata, exist := k.bankKeeper.GetDenomMetaData(ctx, msg.Denom)
		if !exist {
			// create new metadata
			metadata = banktypes.Metadata{
				Base: msg.Denom,
				Name: msg.Denom,
			}
		}
		// update existing metadata
		metadata.Symbol = msg.Symbol
		metadata.Display = strings.ToLower(msg.Symbol)
		if msg.Decimal != 0 {
			metadata.DenomUnits = []*banktypes.DenomUnit{
				{
					Denom:    metadata.Base,
					Exponent: 0,
				},
				{
					Denom:    metadata.Display,
					Exponent: msg.Decimal,
				},
			}
		} else {
			metadata.DenomUnits = []*banktypes.DenomUnit{
				{
					Denom:    metadata.Base,
					Exponent: 0,
				},
			}
		}
		k.bankKeeper.SetDenomMetaData(ctx, metadata)
	} else {
		if len(msg.Contract) == 0 {
			// delete existing mapping
			k.DeleteExternalContractForDenom(ctx, msg.Denom)
		} else {
			if !common.IsHexAddress(msg.Contract) {
				return errors.Wrapf(sdkerrors.ErrInvalidAddress, "invalid contract address (%s)", msg.Contract)
			}
			// update the mapping
			contract := common.HexToAddress(msg.Contract)
			if err := k.ensureContractCode(ctx, contract); err != nil {
				return err
			}
			if err := k.SetExternalContractForDenom(ctx, msg.Denom, contract); err != nil {
				return err
			}
		}
	}

	return nil
}
```
